   задача 6 и натискам enter.
   Така си създавам папка Migrations с клас Configuration
7. В класа ChatRoomContext в конструктора вкарвам следният текст 
   
   Database.SetInitializer(new MigrateDatabaseToLatestVersion<PhonebookContext,Configuration>());
   
   и в същият клас оверайдвам метода 
        
        protected override void OnModelCreating(DbModelBuilder modelBuilder)
        {
            modelBuilder.Conventions.Remove<OneToManyCascadeDeleteConvention>();
            base.OnModelCreating(modelBuilder);
        }
